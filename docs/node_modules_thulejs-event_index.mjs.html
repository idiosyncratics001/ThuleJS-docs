/**
 * @module ThuleJS-event
 * @author Raymon van Dolder
 * @version 1.0.1
 * @example
 * Event levels:
 * 60: FATAL
 * 50: ERROR
 * 40: WARN
 * 30: INFO
 * 20: DEBUG
 * 10: TRACE
 */

import * as event from './src/eventhandler.mjs'


export default{
    /**
     * @description Initialize optional settings for ThuleJS-event by passing an options object
     * @param {object} options Options object
     * @param {boolean} options.console Activate hook on console.log
     * @param {boolean} options.format Activate hook on global Error classestest
     * @example
     * Show usecase here
     */
    init(options){
        return event.init(options);
    },

    
    getMeta(){
        return event.getMeta()
    },

    /**
     * @description Get a new event tag (event format) for a process. </br>
     * EventTag formats the event, adds meta and methods. </br>
     * There is no output unless set with event.addResponse(level, type, function).
     * @property {method(id)} tag add current process name -or- uuid to the tag
     * @property {(string | number)} tag.id name -or- uuid
     * @property {method(type)} type add the type of the process to the tag, manual set of type will overrule default
     * @property {(string)} type.type event type definition
     * @property {method(args)} data add arguments of the process to the tag or any other data  you wish to pass
     * @property {*} data.args node.arguments | any
     * @property {method(msg)} msg add a message to the tag, manual set of message will overrule default
     * @property {string} msg.msg set event message
     * @property {method()} getTag returns the content of the tag
     * @property {object} getTag.return eventTag properties
     * @property {method(event)} fatal send Fatal level event to eventHandler
     * @property {(object | string )} fatal.event Event | Error | message
     * @property {method(event)} error send Error level event to eventHandler
     * @property {(object | string )} error.event Event | Error | message
     * @property {method(event)} warn send Warning level event to eventHandler
     * @property {(object | string )} warn.event Event | Error | message
     * @property {method(event)} info send Info level event to eventHandler
     * @property {(object | string )} info.event Event | Error | message
     * @property {method(event)} debug send debug level event to eventHandler
     * @property {(object | string )} debug.event Event | Error | message
     * @property {method(event)} trace send trace level event to eventHandler
     * @property {(object | string )} trace.event Event | Error | message
     * 
     * @returns {class} new eventTag | empty evenTag with methods
     * @example
     * // example of the format:
     * level: 'ERROR',
     * timestamp: '1-2-2025 21:05:02',
     * file: '/idiosyncratics/run.js',
     * line: '62',
     * type: 'ReferenceError',
     * message: 'arg is not defined',
     * processID: 11035,
     * parentID: 1996,
     * stack: 'ReferenceError: arg is not defined\n' +
     *   '    at test (file:///idiosyncratics/run.js:62:9)\n' +
     *   '    at file:///idiosyncratics/run.js:71:1\n' +
     *   '    at ModuleJob.run (node:internal/modules/esm/module_job:272:25)\n' +
     *   '    at async onImport.tracePromise.__proto__ (node:internal/modules/esm/loader:552:26)\n' +
     *   '    at async asyncRunEntryPointWithESMLoader (node:internal/modules/run_main:98:5)'
     */
    getEventTag(){
        var tag = new event.eventTag()
        return tag
    },

    /**
     * @description Map object with all the responses listed by Level and Type
     * @returns {object} collection of event responses
     */
    getResponseMap(){
        return event.getResponseMap();
    },

    /**
     * @description Adds a response to the EventMap. Multiple responses for each Level and Type possible.
     * @param {number} level level of the event
     * @param {string} type specific type of the event
     * @param {function} response function that will run as response to the error
     * @returns {boolean}
     */
    addResponse(level, type, response){
        return event.addResponse(level, type, response)
    }
    
}
