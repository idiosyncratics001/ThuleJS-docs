/** @module defaultResponses*/
/*	
*   --[ ThuleJS-event | event responses for eventhandler ]--
*	Author: Raymon van Dolder
*/
import event from 'thulejs-event'

// define your type responses here *these are defaults in this file
// or call a processTag.addResponse(level, type, response) in your code to add the response to the list
// each level has a default and leave it last, top down order!

export default function setResponses(){
// ================
// level 60: Fatal
// ================
event.addResponse(60,'Default', (event) => {
    console.log('eventHandler Fatal', event)
    process.exit(1)
})

// ================
// level 50: Error 
// ================

event.addResponse(50,'SyntaxError', (event) => {
    console.log('eventHandler SyntaxError', event)
})

event.addResponse(50,'ReferenceError', (event) => {
    console.log('eventHandler ReferenceError', event)
})

event.addResponse(50,'SyntaxError', (event) => {
    console.log('eventHandler SyntaxError', event)
})

event.addResponse(50,'TypeError', (event) => {
    console.log('eventHandler TypeError', event)
})

event.addResponse(50,'DatabaseError', (event) => {
    console.log('eventHandler DatabaseError', event)
})

// Error level 50: Default
event.addResponse(50, 'Default', (event) => {
    console.log('eventHandler Error Default', event)   // Default on Error //
}); 




// ================
// level 40: Warning  
// ================


// Warning level 40: Default
event.addResponse(40, 'Default', (event) => {
    console.log('eventHandler Warning Default', event)
}); 




// ================
// level 30: Info  
// ================


// Info level 30: Default
event.addResponse(30, 'Default', (event) => {
    console.log('eventHandler Info Default', event)
}); 




// ================
// level 20:  Debug
// ================


// Debug level 20: Default
event.addResponse(20, 'Default', (event) => {
    console.log('eventHandler Debug Default', event)
}); 




// ================
// level 10:  Trace
// ================


// Trace level 10: Default
event.addResponse(10, 'Default', (event) => {
    console.log('eventHandler Trace Default', event)
}); 




/* Explain yourself:
* Map lookup: O(1) (average time complexity)
* Set iteration: O(n) (where n is the number of actions in the set)
* Function calls: O(1) (minimal overhead)
* So, the overall time complexity is approximately O(n), where n is the number of actions in the set. 
* This is a reasonable trade-off for the simplicity and flexibility of the implementation.
**/

// leave this here
}//end defaultResponses
