ThuleJS-env

Managing environment variables and distribution.
ThuleJS-env is a class object cached by Node.js module cache

Description:

  • Managing environment variables and distribution.
    ThuleJS-env is a class object cached by Node.js module cache

Source:
  • node_modules/thulejs-env/index.mjs, line 1
Version:
  • 1.0.5
Author:
  • Raymon van Dolder
Example
import env from 'thulejs-env';

const envOptions = {
  file: '.env',
  path: __dirname,
  strict: true,
  node_env: 'development',
  checkIn: true,
  processEnv: true,
  global: false,
  intrinsics: false
}
env.init(envOptions);

// get the configuration
env.getMeta();

// get the env variables.
_env = env.getEnv();

var timezone = _env.NODE_TZ

// -or-

var timezone = env.getEnv('NODE_TZ')
